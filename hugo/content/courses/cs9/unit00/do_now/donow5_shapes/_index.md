```python
#############
# donow5.py
#############

from turtle import *


def square():
    for i in range(4):
        forward(100)
        right(360/4)

def pentagon():
    for i in range(5):
        forward(100)
        right(360/5)

def hexagon():
    for i in range(6):
        forward(100)
        right(360/6)

def heptagon():
    for i in range(7):
        forward(100)
        right(360/7)

```
